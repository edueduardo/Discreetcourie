function calculateDeliveryPrice(params: {
  type: 'basic' | 'confidential' | 'shopping',
  distance: number,
  datetime: DateTime,
  isRecurring: boolean
}): number {
  
  let basePrice = 0;
  
  // TYPE
  switch(params.type) {
    case 'basic': basePrice = 35; break;
    case 'confidential': basePrice = 55; break;
    case 'shopping': basePrice = 75; break; // Por hora
  }
  
  // DISTANCE
  if (params.distance > 10) {
    basePrice += (params.distance - 10) * 2; // $2 por milha extra
  }
  
  if (params.distance > 25) {
    basePrice += 25; // Fora da zona, surcharge adicional
  }
  
  // TIME
  const hour = params.datetime.hour;
  const day = params.datetime.day;
  
  // After hours (antes 9am ou depois 6pm)
  if (hour < 9 || hour >= 18) {
    basePrice *= 1.3; // +30%
  }
  
  // Weekend
  if (day === 0 || day === 6) {
    basePrice *= 1.5; // +50%
  }
  
  // RECURRING DISCOUNT
  if (params.isRecurring) {
    basePrice *= 0.9; // -10% discount
  }
  
  return Math.round(basePrice);
}
